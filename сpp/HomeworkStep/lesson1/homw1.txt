#include<iostream>
#include <cmath>
using namespace std;
/*
Пользователь вводит координты 2 окружностей и радиус.
Определить - в каком отношении находятся эти окружности.
Формулу нахожения расстояния между двумя точками на плоскости ищем в интернете!
1. Ввести данные;
2. Вычислить расстояние между центрами;
3. Две окружности пересекаются тогда, когда расстояния между центрами меньше суммы их радиусов, 
но больше модуля их разности.
*/
void main()
{
	setlocale(LC_ALL, "Rus");
	int x1, y1, r1, x2, y2, r2;
	int dist = 0;
	cout << "Введите х1 ";cin >> x1;
	cout << "Введите у1 ";cin >> y1;
	cout << "Введите r1 ";cin >> r1;
	cout << "Введите х2 ";cin >> x2;
	cout << "Введите у2 ";cin >> y2;
	cout << "Введите r2 ";cin >> r2;
	if (x1 == x2 && y1 == y2 && r1 == r2) 
	{
		cout << "Окружности совпадают";
	} 
	else 
	{
		dist = sqrt(pow((x1 - x2), 2) + pow((y1 - y2), 2));
		if (dist < (r1 + r2) && dist>abs(r1-r2)) 
		{
			cout << "Окружности пересекаются";
		}
		else 
		{
			cout << "Окружности не пересекаются";
		}
	}
}

		}
	}
